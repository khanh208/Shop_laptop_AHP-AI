nếu bị lỗi thì thử dán đoạn code sau vào Postgre rồi chạy lại 

WITH ranked AS (
    SELECT
        id,
        ROW_NUMBER() OVER (
            PARTITION BY
                COALESCE(NULLIF(TRIM(laptop_name), ''), 'Unknown Laptop'),
                COALESCE(NULLIF(TRIM(model_code), ''), '')
            ORDER BY id
        ) AS rn
    FROM stg_laptop_data
)
DELETE FROM stg_laptop_data s
USING ranked r
WHERE s.id = r.id
  AND r.rn > 1;
